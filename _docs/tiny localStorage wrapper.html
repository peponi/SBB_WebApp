var stor = {

  get : function(id)

  {

  	return JSON.parse(l.getItem(id));

  },

  set : function(id,data)

  {

  	return l.setItem(id, JSON.stringify(data) );

  },

  rm : function(id)

  {

  	l.removeItem(id);

  },

  getKeysFor : function(prefix)

  {

  	var ret = [];

  

  	for (var k in l){

  	   ret.push(k);

  	}

  

  	return ret.filter(function(c){return c.substring(0,prefix.length) == prefix}); 

  }

};
